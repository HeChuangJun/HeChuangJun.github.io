//添加浏览记录
//获得客户端携带cookie—获得名字是pids的cookie
String pids = pid;
Cookie[] cookies = request.getCookies();
if(cookies!=null){
for(Cookie cookie : cookies){
if(“pids”.equals(cookie.getName())){
pids = cookie.getValue();
//1-3-2 本次访问商品pid是8—–>8-1-3-2
//1-3-2 本次访问商品pid是3—–>3-1-2
//1-3-2 本次访问商品pid是2—–>2-1-3
//将pids拆成一个数组
String[] split = pids.split(“-“);//{3,1,2}
List asList = Arrays.asList(split);//[3,1,2]
LinkedList list = new LinkedList(asList);//[3,1,2]
//判断集合中是否存在当前pid
if(list.contains(pid)){
//包含当前查看商品的pid
list.remove(pid);
list.addFirst(pid);
}else{
//不包含当前查看商品的pid 直接将该pid放到头上
list.addFirst(pid);
}
//将[3,1,2]转成3-1-2字符串
StringBuffer sb = new StringBuffer();
for(int i=0;i<list.size()&&i<7;i++){
sb.append(list.get(i));
sb.append(“-“);//3-1-2-
}
//去掉3-1-2-后的-
pids = sb.substring(0, sb.length()-1);
}
}

spring的java配置多数据源

package match2.springboot.application;
import java.beans.PropertyVetoException;
import java.io.IOException;
import javax.sql.DataSource;
import match.service.task.CdjjDataConvertor;
import org.apache.ibatis.session.SqlSessionFactory;
import org.mybatis.spring.SqlSessionFactoryBean;
import org.mybatis.spring.annotation.MapperScan;
import org.mybatis.spring.mapper.MapperScannerConfigurer;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.boot.autoconfigure.jdbc.DataSourceBuilder;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Primary;
import org.springframework.context.annotation.PropertySource;
import org.springframework.core.io.support.PathMatchingResourcePatternResolver;
import org.springframework.jdbc.core.JdbcTemplate;
import com.mchange.v2.c3p0.ComboPooledDataSource;
@Configuration
@PropertySource(value= {
“classpath:jdbc.properties”,
“classpath:barcodepage.properties”,
“classpath:license.properties”,
“classpath:artmis-picture.properties”},
ignoreResourceNotFound=true
)
//扫描mapper
@MapperScan(basePackages=”match2.persistence.**”)
public class SpringBootDataSourceConfig {
//spring4.0特性
@Value(“${match2Mysql.jdbc.driverClassName}”)
private String driverClass;
@Value(“${match2Mysql.jdbc.url}”)
private String jdbcUrl;
@Value(“${match2Mysql.jdbc.username}”)
private String user;
@Value(“${match2Mysql.jdbc.password}”)
private String password;

@Bean(destroyMethod=”close”,name=”match2MySqlDataSource”)
//优先的数据源
@Primary
public ComboPooledDataSource getComboPooledDataSource() throws PropertyVetoException{
ComboPooledDataSource cd=new ComboPooledDataSource();
cd.setDriverClass(driverClass);
cd.setJdbcUrl(jdbcUrl);
cd.setUser(user);
cd.setPassword(password);
cd.setMinPoolSize(5);
cd.setMaxPoolSize(30);
cd.setMaxIdleTime(1800);
cd.setAcquireIncrement(2);
cd.setMaxStatements(1000);
cd.setInitialPoolSize(5);
cd.setIdleConnectionTestPeriod(60);
cd.setAcquireRetryAttempts(30);
cd.setBreakAfterAcquireFailure(false);
cd.setTestConnectionOnCheckout(true);
return cd;
}

//spring4.0特性，注册c3p0连接池
@Value(“${hisOracle.jdbc.driverClassName}”)
private String hisOracledriverClass;
@Value(“${hisOracle.jdbc.url}”)
private String hisOraclejdbcUrl;
@Value(“${hisOracle.jdbc.username}”)
private String hisOracleuser;
@Value(“${hisOracle.jdbc.password}”)
private String hisOraclepassword;
@Bean(destroyMethod=”close”,name=”hisOracleDataSource”)
public ComboPooledDataSource getHisOracleComboPooledDataSource() throws PropertyVetoException{
ComboPooledDataSource cd=new ComboPooledDataSource();
cd.setDriverClass(hisOracledriverClass);
cd.setJdbcUrl(hisOraclejdbcUrl);
cd.setUser(hisOracleuser);
cd.setPassword(hisOraclepassword);
return cd;
}
//注册mysqlJdbcTemplate
@Bean(name=”mysqlJdbcTemplate”)
public JdbcTemplate getJdbcTemplate() throws PropertyVetoException{
JdbcTemplate jt=new JdbcTemplate();
jt.setDataSource(getHisOracleComboPooledDataSource());
return jt;
}
//配置数据对接的实现类
@Value(“${hisOracle.jdbc.retrieveDays}”)
private String retrieveDays;
@Value(“${hisOracle.jdbc.covertsql}”)
private String covertsql;
@Value(“${hisOracle.jdbc.timeSql}”)
private String timeSql;
@Bean(name=”cdjjDataConvertor”)
public CdjjDataConvertor getCdjjDataConvertor(){
CdjjDataConvertor cc=new CdjjDataConvertor();
cc.setCovertsql(covertsql);
cc.setRetrieveDays(Integer.parseInt(retrieveDays));
cc.setTimeSql(timeSql);
return cc;
}

@Bean(name=”sqlSessionFactory”)
public SqlSessionFactoryBean getSqlSessionFactoryBean() throws PropertyVetoException, IOException{
SqlSessionFactoryBean sf = new SqlSessionFactoryBean();
sf.setDataSource(getComboPooledDataSource());
sf.setConfigLocation(new PathMatchingResourcePatternResolver().getResource(“classpath:mybatis-configuration.xml”));
return sf;
}
@Bean(name=”mapperScanner”)
public MapperScannerConfigurer mapperScannerConfigurer() throws PropertyVetoException, IOException {
MapperScannerConfigurer mapperScannerConfigurer = new MapperScannerConfigurer();
mapperScannerConfigurer.setBasePackage(“match2.persistence.**”);
return mapperScannerConfigurer;
}
}
